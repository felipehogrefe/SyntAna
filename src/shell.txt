class sSort{
	method main_[]{
		int n;
		n=read[int];
		a=array[int,n];
		for[i=0;n-1;1]{
			array[i]=read[int];
		}
		a=shellSort[a];
		print[a];			
	}
	method static shellSort_[array[] a]{
		int t = a.length;
		int h = 1;
		while[h<t]{
			h = h * 3 + 1;
		}
		h = h / 3;
		int c;
		int j;
		while[h>0]{
			for[int i = h ;t-1;1]{
				c=a[i];
				j=i;
				while[j >= h && a[j-h] > c]{
					a[j] = a[j - h];
					j = j - h;
				}
				a[j] = c;
			}
			h = h / 2;
		}
	}
}
